```python
# backend/data/rate_limiter.py
import time
from functools import wraps

def rate_limit(calls_per_minute=200):
    """Tushare/AKShare接口限流装饰器"""
    min_interval = 60.0 / calls_per_minute
    def decorator(func):
        last_call = [0.0]
        @wraps(func)
        def wrapper(*args, **kwargs):
            elapsed = time.time() - last_call[0]
            if elapsed < min_interval:
                time.sleep(min_interval - elapsed)
            result = func(*args, **kwargs)
            last_call[0] = time.time()
            return result
        return wrapper
    return decorator
```

**各因子模块实现要点**：

**2.1 value_factors.py (V_*, 12个)**：
- V_EP: `1 / pe_ttm` (从daily_basic取pe_ttm)
- V_BP: `1 / pb` (从daily_basic取pb)
- V_FCFY: `free_cashflow(TTM) / total_mv` (需cashflow+daily_basic)
- V_OCFY: `n_cashflow_act(TTM) / total_mv`

### 阶段3：因子预处理层

**前置依赖**：阶段2
**目标**：实现因子预处理流水线

| 序号 | 任务 | 产出文件 | 核心方法 | 文档参考 |
|------|------|----------|----------|----------|
| 3.1 | 因子预处理流水线 | `backend/preprocessing/preprocess.py` | `preprocess_factors()` | 文档4.1 |

**实现规范**（分块截面预处理，解决OOM问题）：

> **OOM风险分析**：5000只股票 × 2430个交易日(10年) × 208因子列 × 8字节(float64) ≈ **19.2GB**，
> 加Pandas索引开销约**29GB**，16G/32G内存必OOM。
> 
> **核心思路**：预处理的三步操作（缺失值填充、MAD去极值、Z-Score标准化）都是**截面操作**——
> 按trade_date分组，每个截面仅5000行×208列≈8MB，完全不会OOM。
> 因此改为**按交易日逐期从DuckDB读取→处理→写回**，内存峰值仅~8MB。

```python
import duckdb
import numpy as np
import pandas as pd
from loguru import logger

# 维度前缀（用于识别因子列）
DIMENSION_PREFIXES = ['V_', 'P_', 'G_', 'Q_', 'M_', 'S_', 'E_', 
                      'I_', 'C_', 'TP_', 'EV_', 'MF_', 
                      'VOL_', 'SENT_', 'CAL_', 'GOV_',
                      'ANALYST_', 'SECTOR_', 'CONCEPT_', 'TECH_', 'MICRO_',
                      'FQ_', 'DIV_', 'CORP_', 'QV_', 'MS_', 'IM_']

def _identify_factor_cols(columns):
    """识别因子列（以维度前缀开头的数值列）"""
    return [c for c in columns if any(c.startswith(p) for p in DIMENSION_PREFIXES)]

def _fillna_cross_section(df: pd.DataFrame, factor_cols: list) -> pd.DataFrame:
    """截面缺失值填充：按行业内中位数填充"""
    for col in factor_cols:
        df[col] = df.groupby('industry')[col].transform(
            lambda x: x.fillna(x.median())
        )
    return df

def _winsorize_mad_cross_section(df: pd.DataFrame, factor_cols: list, n: float = 3.0) -> pd.DataFrame:
    """截面MAD去极值：对每个因子列整体做MAD截尾（截面内）"""
    for col in factor_cols:
        series = df[col]
        median = series.median()
        mad = (series - median).abs().median()
        upper = median + n * 1.4826 * mad
        lower = median - n * 1.4826 * mad
        df[col] = series.clip(lower, upper)
    return df

def _zscore_cross_section(df: pd.DataFrame, factor_cols: list) -> pd.DataFrame:
    """截面Z-Score标准化：按trade_date分组（单期调用时整表即一个截面）"""
    for col in factor_cols:
        mean = df[col].mean()
        std = df[col].std()
        df[col] = (df[col] - mean) / (std + 1e-8)
    return df

def preprocess_single_date(conn, trade_date: str, table: str = 'factor_raw',
                           output_table: str = 'factor_preprocessed') -> int:
    """单期截面预处理（内存峰值：5000行×208列≈8MB）
    
    从DuckDB读取单期因子数据 → 缺失值填充 → MAD去极值 → Z-Score标准化 → 写回
    """
    # 读取单期数据（~8MB，不会OOM）
    df = conn.execute(
        f"SELECT * FROM {table} WHERE trade_date = '{trade_date}'"
    ).fetchdf()
    
    if df.empty:
        return 0
    
    factor_cols = _identify_factor_cols(df.columns)
    
    # 三步预处理（全部在截面内操作）
    df = _fillna_cross_section(df, factor_cols)
    df = _winsorize_mad_cross_section(df, factor_cols)
    df = _zscore_cross_section(df, factor_cols)
    
    # 写回DuckDB
    df.to_sql(output_table, conn, if_exists='append', index=False)
    return len(df)

def preprocess_factors_chunked(conn, start_date: str = None, end_date: str = None,
                               table: str = 'factor_raw',
                               output_table: str = 'factor_preprocessed',
                               chunk_size: int = 50) -> dict:
    """分块截面预处理主入口（解决OOM问题）
    
    按交易日逐期处理，每期仅加载5000行×208列≈8MB到内存。
    支持分块提交（每chunk_size期提交一次DuckDB写入）。
    
    Args:
        conn: DuckDB连接
        start_date/end_date: 处理日期范围
        table: 原始因子表名
        output_table: 预处理后输出表名
        chunk_size: 每多少期提交一次（减少DuckDB事务开销）
    
    Returns:
        {'total_dates': N, 'total_rows': M, 'elapsed_sec': T}
    """
    import time
    t0 = time.time()
    
    # 获取待处理的交易日列表
    where = []
    if start_date:
        where.append(f"trade_date >= '{start_date}'")
    if end_date:
        where.append(f"trade_date <= '{end_date}'")
    where_clause = f"WHERE {' AND '.join(where)}" if where else ""
    
    dates_df = conn.execute(
        f"SELECT DISTINCT trade_date FROM {table} {where_clause} ORDER BY trade_date"
    ).fetchdf()
    trade_dates = dates_df['trade_date'].tolist()
    
    # 检查已处理日期（断点续传）
    try:
        done_df = conn.execute(
            f"SELECT DISTINCT trade_date FROM {output_table}"
        ).fetchdf()
        done_dates = set(done_df['trade_date'].tolist())
        trade_dates = [d for d in trade_dates if d not in done_dates]
        logger.info(f"跳过已处理{len(done_dates)}期，剩余{len(trade_dates)}期")
    except Exception:
        pass  # 输出表不存在，全量处理
    
    total_rows = 0
    chunk_rows = 0
    
    for i, trade_date in enumerate(trade_dates):
        n = preprocess_single_date(conn, trade_date, table, output_table)
        total_rows += n
        chunk_rows += n
        
        # 分块提交日志
        if (i + 1) % chunk_size == 0:
            logger.info(f"已处理 {i+1}/{len(trade_dates)} 期，本块{chunk_rows}行")
            chunk_rows = 0
    
    elapsed = time.time() - t0
    logger.info(f"预处理完成: {len(trade_dates)}期, {total_rows}行, 耗时{elapsed:.1f}秒")
    return {'total_dates': len(trade_dates), 'total_rows': total_rows, 'elapsed_sec': elapsed}
```

**内存占用对比**：

| 方案 | 内存峰值 | 16G可用 | 32G可用 |
|------|---------|---------|---------|
| 原方案：全量读入factor_df | ~29GB | ❌ OOM | ❌ 极可能OOM |
| 新方案：按trade_date逐期处理 | ~8MB/期 | ✅ | ✅ |
| 新方案：按trade_date逐期+行业中性化 | ~8MB/期 | ✅ | ✅ |

**关键设计决策**：
- 截面操作（Z-Score、MAD）天然按trade_date分组，逐期处理无语义差异
- 缺失值填充按industry分组，单期内约5000/90≈55只/行业，计算量极小
- 行业中性化（如需）同样在截面内完成，不增加内存压力
- 断点续传：检查output_table已有日期，跳过已处理期
- DuckDB写入：单期append开销极小，chunk_size=50仅控制日志频率

## 六、关键设计决策

| 编号 | 决策 | 说明 |
|------|------|------|
| D1 | DuckDB存储 | 不预先建表，由 `pandas.to_sql(if_exists='append')` 自动创建，保留数据源全部字段 |
| D2 | 因子计算架构 | 每个维度一个文件，导出 `register()` + `compute()` 函数；FactorEngine 通过注册表统一调度 |
| D3 | 双源优先级 | Tushare(主) > AKShare(辅) > 计算衍生；北向资金双源互补 |
| D4 | 限流策略 | Tushare高频500次/分(sleep 0.12s)、中频200次/分(sleep 0.3s)、stk_factor_pro 30次/分(sleep 2.0s)；AKShare建议≤60次/分(sleep 1.0s) |
| D5 | 5000积分降级 | 使用 `income`/`balancesheet`/`cashflow`/`fina_indicator` 替代VIP接口 |
| D6 | 因子预处理顺序与分块策略 | 缺失值填充(行业中位数) → MAD去极值(3倍) → Z-Score标准化(截面) → 行业中性化(可选)；**按trade_date逐期从DuckDB读取→处理→写回**，避免全量宽表OOM(29GB) |
| D7 | 市场环境识别 | 信号强度评分机制(0-100分)，强信号(>60)优先，避免硬优先级链屏蔽信号 |
| D8 | IC淘汰 | |IC均值|<0.02 或 IC_IR<0.3 持续3个月则淘汰 |
| D9 | 复合因子依赖 | QV_依赖V_+P_；MS_依赖M_+SENT_；IM_依赖I_+MF_；需按依赖顺序计算 |
| D10 | 日志 | 使用loguru替代标准logging，统一日志格式 |
| D11 | **stk_factor_pro技术面因子** | 接入Tushare `stk_factor_pro`接口(265字段全量存储)，M_/TP_/TECH_维度基础指标优先从该表取值，形态识别仍需本地daily序列计算 |

## 七、风险与注意事项

| 编号 | 风险 | 缓解措施 |
|------|------|----------|
| R1 | AKShare接口不稳定 | safe_call装饰器+自动重试+降级返回NaN |
| R2 | Tushare限流 | rate_limit装饰器+分页获取+sleep间隔 |
| R3 | 因子计算性能 | 批量读取DuckDB后pandas向量化计算，避免逐行查询 |
| R3.1 | **预处理OOM** | 全量宽表(5000股×2430日×208因子≈29GB)直接读入必OOM | **按trade_date逐期截面处理**，单期仅~8MB；详见D6分块策略 |
| R4 | 数据缺失 | 所有因子函数需处理缺失值，返回np.nan而非抛异常 |
| R5 | VOL_IDIOSYNCRATIC | Fama-French三因子回归实现复杂，可先简化为市场单因子回归 |
| R6 | MICRO_*因子 | 依赖实时盘口/tick数据，AKShare接口可能有限制，可标记为可选因子 |
| R7 | SENT_NEWS_SENTIMENT | 新闻舆情需NLP分析，先留placeholder返回np.nan |
| R8 | TP_VCP_PATTERN | VCP杯柄形态识别算法复杂，需仔细实现2-4次收缩的递减幅度判断 |
| R9 | 30套策略硬性过滤 | 每个策略过滤条件不同，需逐一实现并测试 |
| R10 | DuckDB并发 | DuckDB单进程写入，增量同步和因子计算不可同时写同一表 |
| R11 | **stk_factor_pro限频低** | 5000积分仅30次/分钟，历史回补耗时较长 | 按trade_date批量获取(每天1次调用)；历史回补时sleep(2.0)；可考虑多日并行+限流 |
| R12 | **stk_factor_pro字段多** | 265个字段，DuckDB表较宽，查询可能较慢 | 全量存储不裁剪；查询时只SELECT需要的字段；可建视图按需投影 |

## 九、执行建议

1. **按阶段顺序执行**，每个阶段完成后再进入下一阶段
2. **阶段2是核心**，建议按因子维度复杂度从低到高实现：CAL_(最简单) → V_/P_/S_ → G_/Q_/I_ → M_/E_/C_ → EV_/MF_ → TP_/VOL_ → 复合因子
3. **文档中有完整代码的模块优先**：full_sync.py、incremental_sync.py、akshare_sync.py、quality_check.py、preprocess.py、strategy_combos.py、market_regime.py 均有文档代码可直接移植
4. **每完成一个模块立即测试**，不要等全部完成再测试
5. **AKShare接口建议先做smoke test**，确认接口可用再集成
6. **DuckDB建议先用小数据集验证**，确认表结构正确后再全量同步
