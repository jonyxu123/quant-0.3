 ─────────────────────────────────────────────────────────────────────────────────────

                    ▼
              StrategyEngine.score_stocks(strategy_id, factor_df)
                    │
                    ├─▶ 获取策略权重: get_strategy_combo(strategy_id)
                    ├─▶ 加权打分: score = Σ(weight_i × factor_i)
                    ├─▶ 硬性过滤: 应用策略特定过滤条件
                    └─▶ 排序选股: select_top_n(score_df, n=50)

**限流实现建议**:

```python
import time
from functools import wraps

def rate_limit(calls_per_minute=200):
    """Tushare接口限流装饰器"""
    min_interval = 60.0 / calls_per_minute
    def decorator(func):
        last_call = [0.0]
        @wraps(func)
        def wrapper(*args, **kwargs):
            elapsed = time.time() - last_call[0]
            if elapsed < min_interval:
                time.sleep(min_interval - elapsed)
            result = func(*args, **kwargs)
            last_call[0] = time.time()
            return result
        return wrapper
    return decorator

# 使用示例
@rate_limit(calls_per_minute=500)
def fetch_daily(trade_date):
    return pro.daily(trade_date=trade_date)

@rate_limit(calls_per_minute=200)
def fetch_income(period):
    return pro.income(period=period)
```

## 4. 因子预处理与质量评估标准

### 4.1 标准化方法

> **OOM风险与分块策略**：5000股×2430日(10年)×208因子×8字节≈19.2GB，加Pandas开销约29GB，16G/32G内存必OOM。
> 预处理三步操作（缺失值填充、MAD去极值、Z-Score标准化）均为**截面操作**，天然按trade_date分组。
> 改为**按交易日逐期从DuckDB读取→处理→写回**，单期仅5000行×208列≈8MB，完全不会OOM。

```python
# 维度前缀（用于识别因子列）
DIMENSION_PREFIXES = ['V_', 'P_', 'G_', 'Q_', 'M_', 'S_', 'E_', 
                      'I_', 'C_', 'TP_', 'EV_', 'MF_', 
                      'VOL_', 'SENT_', 'CAL_', 'GOV_',
                      'ANALYST_', 'SECTOR_', 'CONCEPT_', 'TECH_', 'MICRO_',
                      'FQ_', 'DIV_', 'CORP_', 'QV_', 'MS_', 'IM_']

def _identify_factor_cols(columns):
    """识别因子列（以维度前缀开头的数值列）"""
    return [c for c in columns if any(c.startswith(p) for p in DIMENSION_PREFIXES)]

def _fillna_cross_section(df, factor_cols):
    """截面缺失值填充：按行业内中位数填充"""
    for col in factor_cols:
        df[col] = df.groupby('industry')[col].transform(
            lambda x: x.fillna(x.median())
        )
    return df

def _winsorize_mad_cross_section(df, factor_cols, n=3.0):
    """截面MAD去极值"""
    for col in factor_cols:
        series = df[col]
        median = series.median()
        mad = (series - median).abs().median()
        upper = median + n * 1.4826 * mad
        lower = median - n * 1.4826 * mad
        df[col] = series.clip(lower, upper)
    return df

def _zscore_cross_section(df, factor_cols):
    """截面Z-Score标准化（单期调用时整表即一个截面）"""
    for col in factor_cols:
        mean = df[col].mean()
        std = df[col].std()
        df[col] = (df[col] - mean) / (std + 1e-8)
    return df

def preprocess_single_date(conn, trade_date, table='factor_raw',
                           output_table='factor_preprocessed'):
    """单期截面预处理（内存峰值：5000行×208列≈8MB，不会OOM）
    
    从DuckDB读取单期因子数据 → 缺失值填充 → MAD去极值 → Z-Score标准化 → 写回
    """
    df = conn.execute(
        f"SELECT * FROM {table} WHERE trade_date = '{trade_date}'"
    ).fetchdf()
    
    if df.empty:
        return 0
    
    factor_cols = _identify_factor_cols(df.columns)
    df = _fillna_cross_section(df, factor_cols)
    df = _winsorize_mad_cross_section(df, factor_cols)
    df = _zscore_cross_section(df, factor_cols)
    df.to_sql(output_table, conn, if_exists='append', index=False)
    return len(df)

def preprocess_factors_chunked(conn, start_date=None, end_date=None,
                               table='factor_raw',
                               output_table='factor_preprocessed',
                               chunk_size=50):
    """分块截面预处理主入口（解决OOM问题）
    
    按交易日逐期处理，每期仅加载~8MB到内存。
    支持断点续传：检查output_table已有日期，跳过已处理期。
    """
    import time
    t0 = time.time()
    
    # 获取待处理的交易日列表
    where = []
    if start_date:
        where.append(f"trade_date >= '{start_date}'")
    if end_date:
        where.append(f"trade_date <= '{end_date}'")
    where_clause = f"WHERE {' AND '.join(where)}" if where else ""
    
    dates_df = conn.execute(
        f"SELECT DISTINCT trade_date FROM {table} {where_clause} ORDER BY trade_date"
    ).fetchdf()
    trade_dates = dates_df['trade_date'].tolist()
    
    # 断点续传：跳过已处理日期
    try:
        done_df = conn.execute(
            f"SELECT DISTINCT trade_date FROM {output_table}"
        ).fetchdf()
        done_dates = set(done_df['trade_date'].tolist())
        trade_dates = [d for d in trade_dates if d not in done_dates]
        logger.info(f"跳过已处理{len(done_dates)}期，剩余{len(trade_dates)}期")
    except Exception:
        pass
    
    total_rows = 0
    chunk_rows = 0
    for i, trade_date in enumerate(trade_dates):
        n = preprocess_single_date(conn, trade_date, table, output_table)
        total_rows += n
        chunk_rows += n
        if (i + 1) % chunk_size == 0:
            logger.info(f"已处理 {i+1}/{len(trade_dates)} 期，本块{chunk_rows}行")
            chunk_rows = 0
    
    elapsed = time.time() - t0
    logger.info(f"预处理完成: {len(trade_dates)}期, {total_rows}行, 耗时{elapsed:.1f}秒")
    return {'total_dates': len(trade_dates), 'total_rows': total_rows, 'elapsed_sec': elapsed}
```

**内存占用对比**：

| 方案 | 内存峰值 | 16G可用 | 32G可用 |
|------|---------|---------|---------|
| 原方案：全量读入factor_df | ~29GB | ❌ OOM | ❌ 极可能OOM |
| 新方案：按trade_date逐期处理 | ~8MB/期 | ✅ | ✅ |

    # === 策略选择逻辑 ===
    # 如果有强特殊信号（评分>60），优先使用
    if signal_scores:
        best_signal = max(signal_scores, key=signal_scores.get)
        if signal_scores[best_signal] > 60:
            return best_signal
    
    # 标准市场环境映射（A-AD，30套策略全覆盖）
    if is_bull and not is_high_vol:
        if small_strong:
            return 'E'  # 小盘成长
        else:
            return 'F'  # 主升浪
    elif is_bull and is_high_vol:

    elif is_bear:

        # 获取交易日历

            (trade_cal['cal_date'] >= start_date) &

        all_data = []

                               method='multi')
                logger.info(f"已同步至 {batch_dates[-1]}, 累计 {len(df_batch)} 条")
                all_data = []
        
        # 循环结束后写入剩余数据
        if all_data:
            df_batch = pd.concat(all_data, ignore_index=True)
            df_batch.to_sql('daily', self.conn, if_exists='append', index=False,
                           method='multi')
            logger.info(f"最后批次写入, 累计 {len(df_batch)} 条")

        for i, trade_date in enumerate(trade_dates):
            try:
                df = self.pro.stk_factor_pro(trade_date=trade_date)
                if not df.empty:
                    # 全部265字段保存，不裁剪
                    df.to_sql('stk_factor_pro', self.conn, if_exists='append', index=False)
                    logger.info(f"[{i+1}/{total}] {trade_date}: {len(df)}条技术面因子")
                time.sleep(2.0)  # 限频30次/分钟 → sleep(2.0)
            except Exception as e:
                logger.error(f"获取{trade_date}技术面因子失败: {e}")
                time.sleep(5.0)  # 失败后等待更长时间
        
        logger.info("历史技术面因子同步完成")
    
    def sync_historical_financials(self):
        """同步历史财务数据（10年）"""
        logger.info("开始同步历史财务数据...")
        
        # 获取过去40个季度
        periods = []
        for year in range(2016, 2027):
            for quarter in ['0331', '0630', '0930', '1231']:
                periods.append(f'{year}{quarter}')
        
        # 同步利润表
        for period in periods:

                if not df.empty:
                    df.to_sql('income', self.conn, if_exists='append', index=False)

                time.sleep(0.2)
            except Exception as e:

        # 同步财务指标
        for period in periods:

    def sync_historical_fund_portfolio(self):

        logger.info("开始同步历史基金持仓...")
        
        # 同步过去8个季度的基金持仓

                                     if_exists='append', index=False)

            time.sleep(2.0)  # stk_factor_pro限频30次/分钟

            return False

        logger.info(f"批量同步: {start_date} to {end_date}")

            SELECT cal_date FROM trade_cal 
            WHERE is_open=1 AND cal_date >= ? AND cal_date <= ?

                    pass

# backend/data/quality_check.py

class DataQualityChecker:
    """数据完整性检查"""
    
    def __init__(self, db_path='stock_data.duckdb'):
        self.conn = duckdb.connect(db_path)
    
    def check_daily_completeness(self, trade_date):
        """检查日线数据完整性"""
        # 应同步股票数
        total_stocks = pd.read_sql('''
            SELECT COUNT(*) FROM stock_basic WHERE list_date <= ?
        ''', self.conn, params=[trade_date]).iloc[0, 0]
        
        # 实际同步股票数
        synced_stocks = pd.read_sql('''
            SELECT COUNT(DISTINCT ts_code) FROM daily WHERE trade_date = ?
        ''', self.conn, params=[trade_date]).iloc[0, 0]
        
        completeness = synced_stocks / total_stocks if total_stocks > 0 else 0
        
        return {
            'trade_date': trade_date,
            'expected': total_stocks,
            'actual': synced_stocks,
            'completeness': completeness,
            'status': 'OK' if completeness > 0.95 else 'WARNING'
        }
    
    def check_missing_dates(self, start_date, end_date):
        """检查缺失的交易日"""
        expected_dates = pd.read_sql('''
            SELECT cal_date FROM trade_cal 
            WHERE is_open=1 AND cal_date >= ? AND cal_date <= ?
        ''', self.conn, params=[start_date, end_date])['cal_date'].tolist()
        
        actual_dates = pd.read_sql('''
            SELECT DISTINCT trade_date FROM daily 
            WHERE trade_date >= ? AND trade_date <= ?
        ''', self.conn, params=[start_date, end_date])['trade_date'].tolist()
        
        missing = set(expected_dates) - set(actual_dates)
        
        return {
            'expected_count': len(expected_dates),
            'actual_count': len(actual_dates),
            'missing_dates': sorted(list(missing)),
            'missing_count': len(missing)
        }
    
    def generate_report(self, trade_date):
        """生成数据质量报告"""
        report = {
            'daily_check': self.check_daily_completeness(trade_date),
            'timestamp': datetime.now().isoformat()
        }
        
        logger.info(f"数据质量报告: {report}")
        return report

# backend/data/akshare_sync.py
"""
AKShare 辅助数据同步模块 (v7.0新增)
负责拉取Tushare无法覆盖的数据：龙虎榜、融资融券、北向资金、大宗交易等。
"""

import akshare as ak
import pandas as pd
import duckdb
import time
import logging
from datetime import datetime, timedelta

logger = logging.getLogger('akshare_sync')


class AKShareSync:
    """AKShare辅助数据同步器"""
    
    def __init__(self, db_path='stock_data.duckdb', tushare_token=None):
        self.conn = duckdb.connect(db_path)
        self.tushare_token = tushare_token
        self._create_tables()
    
    def _create_tables(self):
        """创建AKShare专用数据表（DuckDB v7.2：不预先定义表结构，由to_sql自动创建以保存所有字段）"""
        logger.info("AKShare DuckDB 模式下不预先创建表，将由 pandas.to_sql 自动创建并保存全部字段")
    
    # ===== 具体接口同步方法 =====
    
    def sync_lhb_detail(self, date=None):
        """同步龙虎榜每日明细"""
        if date is None:
            date = datetime.now().strftime('%Y%m%d')
        
        try:
            df = ak.stock_lhb_detail_daily(date=date)
            if df.empty:
                logger.info(f"{date} 无龙虎榜数据")
                return True
            
            df['date'] = date
            df.to_sql('lhb_detail', self.conn, if_exists='append', index=False)
            logger.info(f"龙虎榜 {date}: {len(df)}条")
            return True
            
        except Exception as e:

            return False
    
    def sync_margin_detail(self, date=None):
        """同步融资融券数据（沪深两市）"""
        if date is None:
            date = datetime.now().strftime('%Y%m%d')
        
        all_data = []
        
        for market, name in [('sh', '沪市'), ('sz', '深市')]:
            try:
                if market == 'sh':
                    df = ak.stock_margin_detail_sh(date=date)
                else:
                    df = ak.stock_margin_detail_sz(date=date)
                
                if not df.empty:
                    df['date'] = date
                    all_data.append(df)
                    
            except Exception as e:
                logger.error(f"融资融券{name}同步失败: {e}")
            
            time.sleep(0.5)  # 避免请求过频
        
        if all_data:
            df_all = pd.concat(all_data, ignore_index=True)
            df_all.to_sql('margin_detail', self.conn, if_exists='append', index=False)
            logger.info(f"融资融券 {date}: {len(df_all)}条")
        
        return True
    
    def sync_north_hold(self):

def safe_call(func, fallback=None, retries=2):
    """
    AKShare接口安全调用装饰器：
    - 自动重试机制
    - 降级返回fallback值
    - 错误日志记录
    """
    def wrapper(*args, **kwargs):
        for i in range(retries):
            try:
                result = func(*args, **kwargs)
                return result
            except Exception as e:
                logging.warning(f"调用失败({i+1}/{retries}): {func.__name__} -> {e}")
                time.sleep(1 * (i + 1))
        logging.error(f"调用最终失败，使用降级值: {func.__name__}")
        return fallback
    return wrapper
